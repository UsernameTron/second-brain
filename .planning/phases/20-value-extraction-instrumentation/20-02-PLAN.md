    /**
     * Return YYYY-MM-DD for the supplied Date (or current time) in America/Chicago.
     * @param {Date} [now=new Date()] - Date instance to convert
     * @returns {string} Date key in `YYYY-MM-DD` form, Chicago calendar day

    function dateKey(now = new Date()) {
      const fmt = new Intl.DateTimeFormat('en-CA', {
        timeZone: 'America/Chicago',
        year: 'numeric',
        month: '2-digit',
        day: '2-digit',
      });
      // en-CA produces YYYY-MM-DD natively; no manual reformatting needed.
      return fmt.format(now);
    }
