    **Auth error mapping** (addresses review concern: auth error taxonomy):
    All three functions must catch errors and classify them using AUTH_ERRORS from auth-helper:
    ```js
    const { getAuthenticatedClient, AUTH_ERRORS } = require('./auth-helper');

    function classifyError(err) {
      const msg = err.message || '';
      const code = err.code || err.status;
      // No token in Keychain or Keychain unavailable
      if (msg.includes('AUTH_REQUIRED') || msg.includes('No refresh token')) {
        return { errorType: AUTH_ERRORS.AUTH_REQUIRED, message: msg };
      }
      // Token expired, revoked, or invalid_grant
      if (msg.includes('invalid_grant') || msg.includes('Token has been expired') || code === 401) {
        return { errorType: AUTH_ERRORS.TOKEN_REFRESH_FAILED, message: 'Refresh token expired or revoked. Re-run: node scripts/auth.js' };
      }
      // Insufficient scopes or API not enabled
      if (msg.includes('insufficient') || msg.includes('accessNotConfigured') || code === 403) {
        return { errorType: AUTH_ERRORS.PERMISSION_DENIED, message: msg };
      }
      // Generic error
      return { errorType: 'UNKNOWN', message: msg };
    }
    ```
    Each function's catch block returns: `{ content: [{ type: 'text', text: JSON.stringify({ error: true, ...classifyError(err) }) }] }`

    **getMessageBody(args)** (addresses review concern: MIME parsing policy):
    - Call `getAuthenticatedClient()`
    - Call `gmail.users.messages.get({ userId: 'me', id: args.messageId, format: 'full' })`
    - MIME parsing policy (priority order):
      1. Search `payload.parts` recursively for `mimeType === 'text/plain'` — use its `body.data`
      2. If no text/plain part found, search for `mimeType === 'text/html'` — decode the `body.data` and strip HTML tags using: `html.replace(/<[^>]+>/g, '').replace(/&nbsp;/g, ' ').replace(/&amp;/g, '&').replace(/&lt;/g, '<').replace(/&gt;/g, '>')` (basic entity decode)
      3. If no parts at all, check `payload.body.data` directly (single-part messages)
      4. If none of the above yields content, return `{ messageId, body: '', bodySource: 'none', from, subject, date }`
    - Include a `bodySource` field in the response: `'text/plain'`, `'text/html-stripped'`, or `'none'` so the consumer knows what it got
    - Base64url-decode using: `Buffer.from(data.replace(/-/g, '+').replace(/_/g, '/'), 'base64').toString('utf8')`
    - Return `{ messageId, body, bodySource, from, subject, date }` wrapped in MCP response shape
    - On error: return error response with classified error type
